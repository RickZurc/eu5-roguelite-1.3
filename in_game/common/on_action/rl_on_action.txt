monthly_country_pulse = {
    on_actions = {
        rl_monthly_action
    }
}

on_game_start = {
    on_actions = {
        delay = { days = 1 }
        rl_startup_action
    }
}

rl_monthly_action = {
    trigger = {}
    effect = {
        if = {
            limit = {
                has_variable = num_advances
            }
            # If num_advances is set, this country has had its initialization
            # We check if they have more advances rn than they had last month
            # If they do, we fire the event, then increment the variable
            if = {
                limit = {
                    num_of_advances_researched > var:num_advances
                }
                change_variable = {
                    name = num_advances
                    add = 1
                }
                set_variable = {
                    name = rl_iterator
                    value = 0
                }
                rl_rand_se = { v = 1 }

                set_variable = {
                    name = rl_event_2
                    value = -1
                }

                set_variable = {
                    name = rl_event_3
                    value = -1
                }

                while = {
                    limit = {
                        OR = {
                            var:rl_event_2 = var:rl_event_1
                            var:rl_event_2 = -1
                        }
                    }
                    rl_rand_se = { v = 2 }
                }

                while = {
                    limit = {
                        OR = {
                            var:rl_event_3 = var:rl_event_1
                            var:rl_event_3 = var:rl_event_2
                            var:rl_event_3 = -1
                        }
                    }
                    rl_rand_se = { v = 3 }
                }

                trigger_event_silently = {
                    id = rl_events.1
                }
            }
        } else = {
            # If num_advances is unset, this country has not been initialized
            set_variable = {
                name = num_advances
                value = num_of_advances_researched
            }
            research_advance = advance_type:dummy_parent_advance        # Add the intitialization tech so the dummy advances work properly
            change_variable = {
                name = num_advances
                add = 1
            }
        }
    }
}

rl_startup_action = {
    trigger = {}
    effect = {
        research_advance = advance_type:dummy_parent_advance
    }
}
