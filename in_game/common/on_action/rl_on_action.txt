monthly_country_pulse = {
    on_actions = {
        rl_monthly_action
    }
}

# Extra roll sources so power is earned through active play, not just waiting on
# research. Winning a great battle rewards aggression; a new age is a milestone
# roll for every country. Each source can be toggled in the Community Mod Menu.
on_great_battle_won = {
    on_actions = {
        rl_battle_won_action
    }
}

on_new_age = {
    on_actions = {
        rl_new_age_action
    }
}

# Register our settings with the Community Mod Framework menu.
cmf_on_mod_registration = {
    on_actions = {
        rl_cmf_register_action
    }
}

rl_cmf_register_action = {
    trigger = {}
    effect = {
        rl_cmf_register = yes
    }
}

rl_monthly_action = {
    trigger = {}
    effect = {
        if = {
            limit = {
                has_variable = num_advances
            }
            # Every advance the country researches grants rl_roll_mtd = 1, so
            # modifier:rl_roll_mtd is the live count of researched advances.
            # When it grows past the last synced count, the player just gained
            # an advance, so we open a selection roll (if enabled in settings).
            if = {
                limit = {
                    modifier:rl_roll_mtd > var:num_advances
                }
                if = {
                    limit = {
                        "global_variable_map(cmm|flag:europa_survivors__roll_on_research)" >= 1
                    }
                    rl_do_roll = yes
                }
                # Always sync the baseline to the current count, even if rolls
                # from research are disabled, so we never build up a backlog.
                set_variable = {
                    name = num_advances
                    value = { add = modifier:rl_roll_mtd }
                }
            }
        } else = {
            # First time we see this country: research the dummy parent (so the
            # per-unlock dummy advances can be granted later) and snapshot the
            # current advance count as the baseline.
            research_advance = advance_type:dummy_parent_advance
            set_variable = {
                name = num_advances
                value = { add = modifier:rl_roll_mtd }
            }
            # Initialize the selection variables so the event options can always
            # safely read var:rl_event_1/2/3, plus rarity/reroll state.
            set_variable = { name = rl_event_1 value = -1 }
            set_variable = { name = rl_event_2 value = -1 }
            set_variable = { name = rl_event_3 value = -1 }
            set_variable = { name = rl_rarity value = 1 }
            set_variable = { name = rl_cursed value = 0 }
            set_variable = { name = rl_curse value = 1 }
            # Starting banked rerolls; more are earned by winning great battles.
            set_variable = { name = rl_reroll_tokens value = 2 }
        }
    }
}

# Winning a great battle grants a banked reroll token to the victor's country
# (scope:actor is the winning army, its owner is the country). Rewards military
# play without flooding the player with upgrades.
rl_battle_won_action = {
    trigger = {}
    effect = {
        if = {
            limit = {
                exists = scope:actor.owner
                "global_variable_map(cmm|flag:europa_survivors__roll_on_battle)" >= 1
            }
            scope:actor.owner = {
                rl_grant_reroll_token = yes
            }
        }
    }
}

# A new age is a milestone: grant a roll (root is the country).
rl_new_age_action = {
    trigger = {}
    effect = {
        if = {
            limit = {
                "global_variable_map(cmm|flag:europa_survivors__roll_on_new_age)" >= 1
            }
            rl_do_roll_legendary = yes
        }
    }
}
